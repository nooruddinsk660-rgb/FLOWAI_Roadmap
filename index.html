backend/vehicle_counter.py — Complete ImplementationPython
from ultralytics import YOLO
import cv2, redis, json, time, logging, os
from datetime import datetime
from pathlib import Path

logging.basicConfig(level=logging.INFO, format='%(asctime)s [P1-Vision] %(message)s')
log = logging.getLogger('vision')

# ── COCO class IDs for vehicles (built into YOLOv8n — no training needed) ──
VEHICLE_CLASSES  = {2: 'car', 3: 'motorcycle', 5: 'bus', 7: 'truck'}
AMBULANCE_CLASS  = 80   # After fine-tuning, ambulance becomes class 80
MAX_CAPACITY     = 60   # Vehicles per lane = density 1.0
EMRG_THRESHOLD   = 0.85 # Min confidence to flag emergency

# ── Load models ────────────────────────────────────────────────────────────
BASE_MODEL  = YOLO('yolov8n.pt')       # General vehicle detection
AMBU_MODEL  = YOLO('models/ambulance_v1.pt') if Path('models/ambulance_v1.pt').exists() \
              else BASE_MODEL              # Fallback to base if not yet trained

# ── Redis connection ────────────────────────────────────────────────────────
r = redis.Redis(
    host=os.getenv('REDIS_HOST', 'localhost'),
    port=int(os.getenv('REDIS_PORT', 6379)),
    decode_responses=True
)

def process_intersection(video_path: str, intersection_id: str):
    """Main loop: read video → detect → count → write Redis. Runs forever."""
    log.info(f"Starting vision pipeline for {intersection_id}")

    while True:   # Outer loop: reconnect if video fails
        cap = cv2.VideoCapture(video_path)
        if not cap.isOpened():
            log.error(f"Cannot open {video_path}, retrying in 5s")
            time.sleep(5); continue

        while True:   # Inner loop: frame-by-frame processing
            ret, frame = cap.read()
            if not ret:   # End of video — loop back to start
                cap.set(cv2.CAP_PROP_POS_FRAMES, 0)
                continue

            try:
                # ── Vehicle detection ──────────────────────────────────────
                results   = BASE_MODEL(frame, conf=0.4, verbose=False)
                cls_ids   = results[0].boxes.cls.tolist()
                v_count   = sum(1 for c in cls_ids if int(c) in VEHICLE_CLASSES)
                density   = round(min(v_count / MAX_CAPACITY, 1.0), 3)

                # ── Ambulance detection (uses fine-tuned model) ────────────
                a_results  = AMBU_MODEL(frame, conf=0.5, verbose=False)
                a_confs    = [float(c) for b, c in
                               zip(a_results[0].boxes.cls, a_results[0].boxes.conf)
                               if int(b) == AMBULANCE_CLASS]
                emrg_conf  = max(a_confs) if a_confs else 0.0
                # Only report if above threshold
                emrg_conf  = emrg_conf if emrg_conf >= EMRG_THRESHOLD else 0.0

                # ── Write to Redis ─────────────────────────────────────────
                payload = {
                    'id':                   intersection_id,
                    'count':                v_count,
                    'density':              density,
                    'emergency_confidence': round(emrg_conf, 3),
                    'ts':                   datetime.now().isoformat()
                }
                r.set(f'{intersection_id}:state', json.dumps(payload), ex=10)
                r.set(f'{intersection_id}:health', datetime.now().isoformat(), ex=15)

            except Exception as e:
                log.warning(f"Frame error at {intersection_id}: {e}")

            time.sleep(1)  # 1 inference per second — sufficient for demo

# ── Run all 6 intersections in parallel using multiprocessing ──────────────
if __name__ == '__main__':
    from multiprocessing import Process

    INTERSECTIONS = [
        ('videos/cp_01.mp4',     'CP_01'),
        ('videos/aiims_01.mp4',  'AIIMS_01'),
        ('videos/ina_01.mp4',    'INA_01'),
        ('videos/saket_01.mp4',  'SAK_01'),
        ('videos/nehru_01.mp4',  'NEHRU_01'),
        ('videos/rohini_01.mp4', 'ROHINI_01'),

        Process(target=process_intersection, args=(v, i), daemon=True)

backend/siren_detector.py — Complete Audio DetectionPython
import tensorflow_hub as hub
import librosa, sounddevice as sd
import numpy as np
import redis, asyncio, os, logging

log = logging.getLogger('audio')

# ── YAMNet: 521-class audio classifier. Google pre-trained. No GPU needed. ──
yamnet   = hub.load('https://tfhub.dev/google/yamnet/1')

# Emergency vehicle siren class indices in YAMNet's output
SIREN_IDX = {397: 'ambulance', 398: 'fire_engine', 399: 'police', 400: 'civil_defense'}
SR         = 16000   # YAMNet requires 16kHz mono
WIN_SEC    = 0.5     # 0.5s classification window
CONSEC     = 3       # Require 3 consecutive hits to avoid false positives

r = redis.Redis(host=os.getenv('REDIS_HOST','localhost'), decode_responses=True)

async def run_audio_detector():
    hit_counter = 0
    log.info("Audio siren detector started — monitoring microphone")

    while True:
        try:
            # Record 0.5-second audio window from default microphone
            audio   = sd.rec(int(WIN_SEC * SR), samplerate=SR, channels=1, dtype='float32')
            sd.wait()
            waveform = audio.flatten()

            # Run YAMNet — returns (frames×521) score matrix
            scores, _, _ = yamnet(waveform)
            mean_s = scores.numpy().mean(axis=0)

            # Get max confidence across all emergency siren classes
            siren_conf = float(max(mean_s[idx] for idx in SIREN_IDX))

            # Consecutive confirmation to filter transient sounds
            if siren_conf > 0.60: hit_counter = min(hit_counter + 1, CONSEC + 2)
            else:                 hit_counter = max(hit_counter - 1, 0)

            confirmed = siren_conf if hit_counter >= CONSEC else 0.0
            r.set('audio:siren_confidence', round(confirmed, 3), ex=3)

            if confirmed > 0.70:
                log.warning(f"🚨 SIREN CONFIRMED — conf={confirmed:.2f} hits={hit_counter}")

        except Exception as e:
            log.warning(f"Audio error: {e}")
            r.set('audio:siren_confidence', 0.0, ex=3)

        await asyncio.sleep(0.1)   # 10 classifications/second


# ── Fusion function imported by signal_brain.py ────────────────────────────
def get_fused_emergency_confidence(intersection_id: str, r: redis.Redis) -> float:
    """Late fusion: weighted average of audio + visual confidence signals."""
    audio  = float(r.get('audio:siren_confidence')                       or 0.0)
    visual = float(r.get(f'{intersection_id}:state')                     or '{"emergency_confidence":0}')
    import json
    visual = json.loads(visual).get('emergency_confidence', 0.0) if isinstance(visual, str) else visual

    # Audio weighted higher — occlusion resistant (ambulance behind bus)
    fused = (audio * 0.55) + (visual * 0.45)

    # High single-modality override: at these levels false positive <2%
    if audio > 0.88 or visual > 0.92:
        fused = max(audio, visual)

    return round(fused, 3)

export default function App() {
  const [intersections, setIntersections] = useState([])
  const [emergency,     setEmergency]     = useState(null)
  const [connected,     setConnected]     = useState(false)
  const [showAQI,       setShowAQI]       = useState(true)
  const wsRef = useRef(null)

  const connect = useCallback(() => {
    const ws = new WebSocket(WS_URL)
    wsRef.current = ws

    ws.onclose = () => {
      setConnected(false)
      console.log('WS disconnected — reconnecting in 3s')

      const data = JSON.parse(evt.data)
      setIntersections(data.intersections || [])
      if (data.emergency) setEmergency(data.emergency)
      else if (!data.emergency) setEmergency(null)
    }
  }, [])

      method: 'POST',
      headers: { 'Content-Type': 'application/json' },

G.add_weighted_edges_from(lambda: [(a, b, {'weight': w}) for a, b, w in EDGES], weight='weight')

    ('CP_01',    'INA_01'):   'south',
    ('INA_01',   'AIIMS_01'): 'south',

def build_routes():
    routes = {}
    node_pairs = [('CP_01', 'AIIMS_01'), ('ROHINI_01', 'AIIMS_01'),

    for origin, dest in node_pairs:

            path = nx.dijkstra_path(G, origin, dest, weight='weight')
            legs = []

                legs.append({
                    'id':        path[i],

        except nx.NetworkXNoPath:
            pass

model = xgb.XGBRegressor()
model.load_model('models/congestion_model.json')

INTERSECTION_ENC = {'CP_01':0, 'AIIMS_01':1, 'INA_01':2, 'SAK_01':3,
                    'NEHRU_01':4, 'KALK_01':5, 'LODHI_01':6, 'ROHINI_01':7}

PEAK_HOURS = {8, 9, 17, 18, 19}
FESTIVAL_DATES = {(3, 14), (10, 21)}  # (month, day) — Holi, Diwali in window

r = redis.Redis(host=os.getenv('REDIS_HOST', 'localhost'), decode_responses=True)

def forecast_density(intersection_id: str, horizon: int = 30) -> list:
    now = datetime.now()
    preds = []
    for offset in range(0, horizon + 1, 5):
        t = now + timedelta(minutes=offset)
        feats = pd.DataFrame([{
            'hour':         t.hour,
            'minute':       t.minute,
            'day_of_week':  t.weekday(),
            'is_weekend':   int(t.weekday() >= 5),
            'is_peak':      int(t.hour in PEAK_HOURS),
            'is_festival':  int((t.month, t.day) in FESTIVAL_DATES),
            'intersection': INTERSECTION_ENC.get(intersection_id, 0),
        }])
        density = float(np.clip(model.predict(feats)[0], 0.0, 1.0))
        level   = 'HIGH' if density > 0.70 else 'MEDIUM' if density > 0.40 else 'LOW'
        preds.append({'t_plus': offset, 'density': round(density, 2), 'level': level})
    return preds

async def run_forecast_loop(intersection_ids: list):
    """Background task: update forecasts in Redis every 60 seconds."""
    while True:
        for iid in intersection_ids:
            preds = forecast_density(iid)
            r.set(f'forecast:{iid}:30min', json.dumps(preds), ex=120)
        await asyncio.sleep(60)
